def outliers(data):
    Q1 = data.quantile(0.25)
    Q3 = data.quantile(0.75)
    IIQ = Q3 - Q1
    lower_bound = Q1 - 1.5*IIQ
    upper_bound = Q3 + 1.5*IIQ
    no_outlier = data[(data > lower_bound) & (data < upper_bound)]
    return no_outlier

Fare_no_outlier = outliers(data['Fare'])

def intervalo_acc(crossval):
    mean = crossval_acc.mean()
    dv = crossval_acc.std()
    print('\nAcurcia mdia: {:.2f}%'.format(mean*100))
    print('Desvio-padro da acurcia: {:.2f}%'.format(dv*100))
    print('Intervalo da acurcia: [{:.2f}% - {:.2f}%]\n'.format((mean-2*dv)*100, (mean + 2*dv)*100))

def intervalo_prec(crossval):
    mean = crossval.mean()
    dv = crossval.std()
    print('\nPreciso mdia: {:.2f}%'.format(mean*100))
    print('Desvio-padro da preciso: {:.2f}%'.format(dv*100))
    print('Intervalo da preciso: [{:.2f}% - {:.2f}%]\n'.format((mean-2*dv)*100, (mean + 2*dv)*100))

def intervalo_rc(crossval):
    mean = crossval.mean()
    dv = crossval.std()
    print('\nRecall mdio: {:.2f}%'.format(mean*100))
    print('Desvio-padro do recall: {:.2f}%'.format(dv*100))
    print('Intervalo do recall: [{:.2f}% - {:.2f}%]\n'.format((mean-2*dv)*100, (mean + 2*dv)*100))

# os modelos so moderados, mas pode melhorar
intervalo_acc(crossval_acc)
